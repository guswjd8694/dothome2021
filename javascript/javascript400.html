01. 1부터 100까지 출력하기 

{
    document.write("*** 1부터 100까지 출력하기 ***")

    for(let i=1; i<=100; i++){
        document.write(i);
    }
}

02. 1부터 100까지 출력하기 : 짝수만 

{
    document.write("*** 1부터 100까지 출력하기 : 짝수만 ***")

    for(let i=1; i<=100; i++){
        if(i % 2 == 0) {
            document.write(i);
        }
    }
}

04. 1부터 100까지 출력하기 : 짝수는 빨간색, 홀수는 파란색 출력 

{
    document.write("*** 1부터 100까지 출력하기 : 짝수는 빨간색, 홀수는 파란색 출력 ***")

    for(let i=1; i<=100; i++){
        if(i % 2 == 0) {
            document.write("<span style='color:red'> " + i + "</span>");
        }else{
            document.write("<span style='color:blue'> " + i + "</span>");
        }
    }
}

05. 1부터 100까지 합 출력하기 

{
    document.write("*** 1부터 100까지 합 출력하기 ***")

    let sum = 0;

    for(let i = 1; i<= 100; i++){
        sum += i;
    }
    document.write(sum)
}

0. 구구단 출력하기 

{    
    document.write("*** 구구단 출력하기 ***")

    for(let i=2; i<=9; i++){
        for(let j=1; j<=9; j++){
            let sum = i * j;
        document.write(i + "*" + j + "=" + sum);
        }
    }
}

0. 테이블 만들기 1 

{
    document.write("*** 테이블 만들기 1 ***")

    let num = 1;
    let table = "<table>";

    for( let i=1; i<=10; i++ ){
        table += "<tr>";
        for( let j=1; j<=10; j++ ){
            table += "<td>"+ num +"</td>";
            num += 1;
        }
        table += "</tr>";
    }

    table += "</table>";    

    document.write(table);
}

0. 테이블 만들기 2 

{
    document.write("*** 테이블 만들기 2 ***")

    let num = 100;
    let table = "<table>";

    for( let i=1; i<=10; i++ ){
        table += "<tr>";
        for( let j=1; j<=10; j++ ){
            table += "<td>"+ num +"</td>";
            num -= 1;
        }
        table += "</tr>";
    }

    table += "</table>";    

    document.write(table);
}
