{
    $(".reset1").click(function(){
        $(".view1 li").removeClass("red");
    });

    $(".btn1-1").click(function(){
        $(".view1 .class").addClass("red");
    });

    $(".btn1-2").click(function(){
        $(".view1 #id").addClass("red");
    });

    $(".btn1-3").click(function(){
        $(".view1 li").addClass("red");
    });

    $(".btn1-4").click(function(){
        $(".view1 .class, .view #id").addClass("red");
    });

    $(".btn1-5").click(function(){
        $(".view1 *").addClass("red");
    });
}
