19. 콜백 함수 : 동기 / 비동기 

{
    document.write("*** 19. 콜백함수 : 동기/비동기 ***");
    
    //01
    function funA(){
        document.write("funA가 실행되었습니다.");
    }
    function funB(){
        document.write("funB가 실행되었습니다.");
    }
    funA();
    funB();

    //02
    function funC(){
        setTimeout(() => {
            //document.write("funC가 실행되었습니다.");
            console.log("C")
        }, 1000);
    }
    function funD(){
        //document.write("funD가 실행되었습니다.");
        console.log("D")
    }
    funC();
    funD();

    document.write("");

    //03
    function funE(callback){
        setTimeout(() => {
            //document.write("funE가 실행되었습니다.");
            console.log("E")
            callback();
        }, 1000);
    }
    function funF(){
        //document.write("funF가 실행되었습니다.");
        console.log("F")
    }
    funE(function(){
        funF();
    });
}

26. 함수 유형 : 객체 생성자 함수 

{
    document.write("*** 26. 함수 유형 : 객체 생성자 함수 ***"); //젤 마니씀 젤 중요한 형태예용가리

    function func(num, name, job){
        this.num = num;
        this.name = name;
        this.job = job;

        this.result = function(){
            document.write(this.num + ". 기요운 " + this.name + "의 직업은 " + this.job + "입니다.");
        }
    }
    const info1 = new func("1", "달래", "멍멍이");         //아~~ new가 드가면 인스턴스 생성을 한거구나 ~ ~~
    const info2 = new func("2", "달랭이", "고구마말랭이");  //값이 있으면 알아서 인스턴스 생성 해줌니당구

    info1.result();
    info2.result();
}

28. 함수 유형 : 객체 리터럴 함수 

{
    document.write("*** 28. 함수 유형 : 객체 리터럴 함수 ***"); //젤 마니씀 젤 중요한 형태예용가리

    function func(num, name, job){
        this.num = num;
        this.name = name;
        this.job = job;
    }
    func.prototype = { //중괄호 요게 리터럴임
        result1 : function(){
            document.write(this.num + ". 기요운 " + this.name + "의 직업은 " + this.job + "입니다.");
        },
        result2 : function(){
            document.write(this.num + ". 기요운 " + this.name + "의 직업은 " + this.job + "입니다.");
        }
    }
    const info1 = new func("1", "달래", "멍멍이");         //아~~ new가 드가면 인스턴스 생성을 한거구나 ~ ~~
    const info2 = new func("2", "달랭이", "고구마말랭이");  //값이 있으면 알아서 인스턴스 생성 해줌니당구
    
    info1.result1();
    info2.result2();
}
