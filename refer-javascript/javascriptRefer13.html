07. 테이블 만들기 1 

{
    document.write("*** 테이블 만들기 1 ***")

    let num = 1;
    let table = "<table>";

    for( let i=1; i<=10; i++ ){
        table += "<tr>";
        for( let j=1; j<=10; j++ ){
            table += "<td>"+ num +"</td>";
            num += 1;
        }
        table += "</tr>";
    }

    table += "</table>";    

    document.write(table);
}
